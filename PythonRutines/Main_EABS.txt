from numpy import unique

def divide_basisset(basisets, symbols):
    global Nbtm, Nmol, Ntop
    basis = {}
    pseudo = {}

    basisets = basisets.split(f'{intro}')
    j = -1
    for i in range(len(basisets)):
        j+=1
        line = basisets[j]
        if line == '' or line == ' ':
            break
        basis[line] = ''
        key = line
        j+=1
        line = basisets[j]
        while not '****' in line:
            basis[key] += line + f'{intro}'
            j+=1
            line = basisets[j]
        
    j+=1
    line = basisets[j]
    while line != '' or line != ' ':
        if line in basis.keys():
            key = line
            pseudo[key] = '' 
        else:
            pseudo[key] += line + f'{intro}'
        j+=1
        line = basisets[j]
        if line == '' or line == ' ':
            break
        
    basiset_molecule = ''
    basiset_electrode = ''
    molecule = unique([elem_dict[x] for x in symbols[Nbtm:Nbtm+Nmol]])
    for key in basis.keys():
        flag = True
        for elem in molecule:
            if elem+' ' in key:
                basiset_molecule += key + f'{intro}'    
                basiset_molecule += basis[key]
                basiset_molecule += f'****{intro}'
                flag = False
        if flag:
            mod_key = [int(x) for x in key.split()]
            for i,x in enumerate(mod_key):
                if x > Nbtm:
                    x = x-Nmol
                basiset_electrode += f'{x} '
            basiset_electrode += f'{intro}'
            basiset_electrode += basis[key]
            basiset_electrode += f'****{intro}'
    basiset_electrode += f'{intro}'
    basiset_molecule += f'{intro}'

    for key in pseudo.keys():
        flag = True
        for elem in molecule:
            if elem+' ' in key:
                basiset_molecule += key + f'{intro}'    
                basiset_molecule += pseudo[key]
                flag = False
        if flag:
            mod_key = [int(x) for x in key.split()]
            for i,x in enumerate(mod_key):
                if x > Nbtm:
                    x = x-Nmol
                basiset_electrode += f'{x} '
            basiset_electrode += f'{intro}'
            basiset_electrode += pseudo[key]
    
    basiset_electrode += f'{intro}{intro}{intro}{intro}'
    basiset_molecule += f'{intro}{intro}{intro}{intro}'
    return basiset_electrode, basiset_molecule

def check_for_subst(header):
    chk = ''
    header = header.split(f'{intro}')
    for i,hd in enumerate(header):
        if not '%Subst L' in hd:
            if i==len(header)-2:
                chk += '0,1'
            else:
                chk += hd+f'{intro}'
    return chk
