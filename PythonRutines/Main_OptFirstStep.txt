
def create_rest_gjf(header, basisets, Elem, Coords, name):
    with open('../'+name+'/input'+name+'.gjf', 'w') as out:
        out.write(header)
        for elem, poss in zip(Elem, Coords):
            out.write(f'{elem}  {float(poss[0]):.6f}   {float(poss[1]):.6f}   {float(poss[2]):.6f} {intro}')
        out.write(f'{intro}')
        out.write(basisets)
    return
